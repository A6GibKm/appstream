A helper to build a Video.

Example

use url::Url;
use appstream::builders::VideoBuilder;

fn main() -> Result<(), url::ParseError> {

    let video = VideoBuilder::new(Url::parse("https://example.com/foobar/screencast.mkv")?)
                .width(1600)
                .height(900)
                .codec("av1")
                .build();

    Ok(())
}
